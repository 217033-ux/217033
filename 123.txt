    <script>
        // Tab switching logic
        function showSection(sectionId) {
            document.querySelectorAll('.content-section').forEach(section => {
                section.classList.add('hidden');
            });
            document.getElementById(sectionId).classList.remove('hidden');
            window.scrollTo(0, 0);
        }

        // Quiz Data
        const quizData = [
            { q: "清朝統治中國內地共多少年？", a: ["168年", "268年", "368年"], c: 1 },
            { q: "哪位皇帝在位61年，平三藩並收復台灣？", a: ["雍正", "康熙", "乾隆"], c: 1 },
            { q: "「攤丁入畝」與「耗羨歸公」是哪位皇帝的改革？", a: ["康熙", "雍正", "嘉慶"], c: 1 },
            { q: "清朝最具代表性的滿漢融合大宴稱作？", a: ["萬壽宴", "鹿鳴宴", "滿漢全席"], c: 2 },
            { q: "清朝晚期引進機器鑄造的銀幣常被稱為？", a: ["交子", "龍洋", "開元通寶"], c: 1 },
            { q: "哪一部清代小說被譽為古典小說的巔峰？", a: ["三國演義", "水滸傳", "紅樓夢"], c: 2 },
            { q: "清朝的大額交易主要使用什麼？", a: ["紙鈔", "銀兩", "銅錢"], c: 1 },
            { q: "清朝末代皇帝是誰？", a: ["光緒", "同治", "溥儀"], c: 2 },
            { q: "鴉片戰爭爆發於哪位皇帝在位期間？", a: ["道光", "咸豐", "光緒"], c: 0 },
            { q: "京劇起源於乾隆年間的什麼活動？", a: ["祭天大典", "徽班進京", "秋獮"], c: 1 }
        ];

        let currentQuestion = 0;
        let score = 0;

        function loadQuestion() {
            const quizContent = document.getElementById('quiz-content');
            const data = quizData[currentQuestion];
            
            quizContent.innerHTML = `
                <div class="mb-6">
                    <span class="text-sm text-gray-500">問題 ${currentQuestion + 1} / 10</span>
                    <h3 class="text-xl font-bold mt-2">${data.q}</h3>
                </div>
                <div class="space-y-3">
                    ${data.a.map((opt, i) => `
                        <button onclick="handleAnswer(${i})" class="w-full text-left p-4 rounded-lg border-2 border-gray-200 hover:border-[#c0a16b] hover:bg-[#fdfaf6] transition">
                            ${opt}
                        </button>
                    `).join('')}
                </div>
            `;
        }

        function handleAnswer(choice) {
            if (choice === quizData[currentQuestion].c) {
                score++;
            }
            
            currentQuestion++;
            
            if (currentQuestion < quizData.length) {
                loadQuestion();
            } else {
                showResult();
            }
        }

        function showResult() {
            document.getElementById('quiz-content').classList.add('hidden');
            const resultDiv = document.getElementById('quiz-result');
            resultDiv.classList.remove('hidden');
            document.getElementById('score-text').innerText = `得分：${score * 10} / 100`;
        }

        function resetQuiz() {
            currentQuestion = 0;
            score = 0;
            document.getElementById('quiz-content').classList.remove('hidden');
            document.getElementById('quiz-result').classList.add('hidden');
            loadQuestion();
        }

        // Initialize quiz
        loadQuestion();
    </script>
